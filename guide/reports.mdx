## Overview

Movilidad provides powerful reporting capabilities for tracking field operations, manager performance, and collection metrics. Reports can be filtered by date range, manager, leader, and project.

<Info>
Reports are available to both leaders (for their team) and admins (for all teams and projects).
</Info>

## Generating Reports

### Report Management Method

The core reporting method provides comprehensive data:

```javascript imports/api/records/recordsMethods.js
Meteor.methods({
  reportManagement: async function({
    page,
    pageSize,
    managerId,
    period,
    date,
    startDate,
    endDate,
    sortField,
    sortOrder = 1,
    leaderId,
    assigneds
  }) {
    let query = {};

    if (leaderId) query.leaderId = leaderId;
    if (assigneds) query.GESTOR = { $exists: true };
    if (managerId) query.GESTOR = managerId;
    if (period) query.period = period;
    
    if (date) {
      const normalizedDate = moment(date, 'DD/MM/YYYY')
        .startOf('day')
        .toDate();
      query.updatedAt = normalizedDate;
    }
    
    if (startDate && endDate) {
      const normalizedStartDate = moment(startDate, 'DD/MM/YYYY')
        .startOf('day')
        .toDate();
      const normalizedEndDate = moment(endDate, 'DD/MM/YYYY')
        .endOf('day')
        .toDate();
      query.updatedAt = {
        $gte: normalizedStartDate,
        $lte: normalizedEndDate
      };
    }

    // Get paginated records
    const pages = await recordsCollection
      .rawCollection()
      .aggregate([
        { $match: query },
        { $sort: { [sortField]: sortOrder } },
        {
          $facet: {
            metadata: [{ $count: 'totalCount' }],
            data: [
              { $skip: (page - 1) * pageSize },
              { $limit: pageSize }
            ]
          }
        },
        {
          $project: {
            data: 1,
            totalCount: { $arrayElemAt: ['$metadata.totalCount', 0] }
          }
        }
      ])
      .toArray();

    // Calculate totals
    const totales = /* aggregation for totals */;
    
    // Calculate performance by manager
    const report = /* aggregation for manager stats */;

    return { ...pages, totales, report };
  }
});
```

```javascript imports/ui/components/reportView/index.jsx
    total: 0
  });

      page,
      pageSize,
      managerId: selectedManager,
      period: period ? moment(period).toISOString() : null,

      startDate,

        setPagination(prev => ({
          ...prev,
          current: page,
          total: result[0].totalCount
        }));
      }
    });
  }
```

## Financial Totals

The totals aggregation provides financial summary:

```javascript
const totales = await recordsCollection
  .rawCollection()
  .aggregate([
    { $match: query },
    {
      $facet: {
        metadata: [
          {
            $group: {
              _id: null,
              deudaTotalAsignada: {
                $sum: { $toDouble: '$DEUDA_TOTAL_ASIGNADA' }
              },
              indicadorCounts: {
                $push: { $toUpper: '$INDICADOR' }
              },
              descripcionTipoProductoCounts: {
                $push: { $toUpper: '$DESCRIPCION_TIPO_PRODUCTO' }
              }
            }
          }
        ]
      }
    },
    {
      $project: {
        deudaTotalAsignada: {
          $ifNull: [
            { $arrayElemAt: ['$metadata.deudaTotalAsignada', 0] },
            0
          ]
        },
        indicadorCounts: {
          Normalizacion: { /* count logic */ },
          Contencion: { /* count logic */ },
          Castigado: { /* count logic */ }
        },
        descripcionTipoProductoCounts: {
          GAS: { /* count logic */ },
          BRILLA: { /* count logic */ },
```

## Manager Performance Report

Performance metrics aggregated by manager:

```javascript
const report = await recordsCollection
  .rawCollection()
  .aggregate([
    { $match: query },
    {
      $group: {
        _id: '$GESTOR',
        asignadas: { $sum: 1 },
        totalDeudaCorrienteAsignada: {
          $sum: { $toDouble: '$DEUDA_TOTAL_ASIGNADA' }
        },
        gestionadas: {
          $sum: {
            $cond: [{ $eq: ['$status', 'reviewed'] }, 1, 0]
          }
        },
        totalDeudaCorrienteGestionada: {
          $sum: {
            $cond: [
              { $eq: ['$status', 'reviewed'] },
              { $toDouble: '$DEUDA_TOTAL_ASIGNADA' },
              0
            ]
          }
        },
        pendientes: {
          $sum: {
            $cond: [{ $eq: ['$status', 'pending'] }, 1, 0]
          }
        }
      }
    },
    {
      $group: {
        _id: null,
        totalAsignadas: { $sum: '$asignadas' },
        totalDeudaCorrienteAsignada: { $sum: '$totalDeudaCorrienteAsignada' },
        totalGestionadas: { $sum: '$gestionadas' },
        totalDeudaCorrienteGestionada: { $sum: '$totalDeudaCorrienteGestionada' },
        totalPendientes: { $sum: '$pendientes' },
        gestores: {
          $push: {
            gestor: '$_id',
            asignadas: '$asignadas',
            totalDeudaCorrienteAsignada: '$totalDeudaCorrienteAsignada',
            gestionadas: '$gestionadas',
            totalDeudaCorrienteGestionada: '$totalDeudaCorrienteGestionada',
            pendientes: '$pendientes'
          }
        }
      }
    }
  ])
  .toArray();
```

### Report Table

```javascript imports/ui/components/records/reportTable.jsx
import { Table } from 'antd';

export default function ReportTable({ report, managers }) {
  const columns = [
    {
      title: 'Manager',
      dataIndex: 'gestor',
      render: (gestorId) => {
        const manager = managers.find(m => m.id === gestorId);
        return manager?.username || gestorId;
      }
    },
    {
      title: 'Assigned',
      dataIndex: 'asignadas'
    },
    {
      title: 'Completed',
      dataIndex: 'gestionadas'
    },
    {
      title: 'Pending',
      dataIndex: 'pendientes'
    },
    {
      title: 'Debt Assigned',
      dataIndex: 'totalDeudaCorrienteAsignada',
      render: (value) => `$${value.toLocaleString()}`
    },
    {
      title: 'Debt Managed',
      dataIndex: 'totalDeudaCorrienteGestionada',
      render: (value) => `$${value.toLocaleString()}`
    }
  ];
  
  return (
    <Table
      columns={columns}
      dataSource={report?.gestores || []}
      rowKey="gestor"
      summary={() => (
        <Table.Summary.Row>
          <Table.Summary.Cell>Total</Table.Summary.Cell>
          <Table.Summary.Cell>{report?.totalAsignadas}</Table.Summary.Cell>
          <Table.Summary.Cell>{report?.totalGestionadas}</Table.Summary.Cell>
          <Table.Summary.Cell>{report?.totalPendientes}</Table.Summary.Cell>
          <Table.Summary.Cell>
            ${report?.totalDeudaCorrienteAsignada.toLocaleString()}
          </Table.Summary.Cell>
          <Table.Summary.Cell>
            ${report?.totalDeudaCorrienteGestionada.toLocaleString()}
          </Table.Summary.Cell>
        </Table.Summary.Row>
      )}
    />
  );
}
```

## Excel Export

Export reports to Excel format:

```javascript
import XLSX from 'xlsx';
import { saveAs } from 'file-saver';

async function handleExport() {
  const pageSize = 100;
  let page = 1;
  const allData = [];
  let total = 10000000;
  
  try {
    // Fetch all pages
    while (total / pageSize >= page) {
      const partData = await Meteor.callAsync('reportManagement', {
        page,
        pageSize,
        managerId: selectedManager,
        leaderId: Meteor.userId(),
        period: period ? moment(period).toISOString() : null,
        date,
        startDate,
        endDate,
        sortField: 'CONTRATO'
      });
      
      if (partData[0].totalCount) total = partData[0].totalCount;
      
      allData.push(partData[0].data);
      setDownloadReport({ 
        percent: page / (total / pageSize), 
        loading: true 
      });
      page++;
    }
    
    // Create worksheet
    const ws = XLSX.utils.json_to_sheet(allData.flat());
    const wb = XLSX.utils.book_new();
    XLSX.utils.book_append_sheet(wb, ws, 'Report');
    
    // Download file
    XLSX.writeFile(wb, `Reporte_${Date.now()}.xlsx`);
  } catch (error) {
    message.error(error.message);
  } finally {
    setDownloadReport({ percent: 0, loading: false });
  }
}
```

<Note>
Large exports are fetched in batches to avoid memory issues. A progress indicator shows download status.
</Note>
